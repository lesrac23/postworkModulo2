/**
 * Ejercicio 4: Getters y Setters Personalizados
 * Implementa una clase Temperatura con un atributo en Celsius. 
 * Incluye getters y setters personalizados para obtener y establecer la temperatura en Fahrenheit.
 */
 class Temperatura(celsius: Double) {//celsius es un parametro
    var celsius = celsius //inicializamos la variable
        set(value) { //
            field = value
            println("Temperatura establecida a $value째C")
        }

    var fahrenheit: Double
        get() = celsius * 9/5 + 32
        set(value) {
            celsius = (value - 32) * 5/9
            println("Temperatura establecida a $value째F")
        }
}

fun main() {
    val temp = Temperatura(25.0)
    println("En Fahrenheit: ${temp.fahrenheit}째F")
    temp.fahrenheit = 68.0
    println("En Celsius: ${temp.celsius}째C")
}
